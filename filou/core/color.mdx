## Material Colors

<Playground>
  <Provider renderer={renderer}>
    <FelaComponent style={{ display: 'flex', flexWrap: 'wrap' }}>
      {materialColors.map((color, j) => (
        <FelaComponent
          style={{
            color: color.palette[7],
            textAlign: 'center',
            display: 'flex',
            flexDirection: 'column'
          }}
          color={color}
          key={color.palette[7]}
          render={({ className }) => (
            <div>
              <h4 className={className}>
                {j}: {color.color}
              </h4>
              {color.palette.map((palette, i) => (
                <FelaComponent
                  rule={({ theme, color, palette }) => ({
                    backgroundColor: color.palette[palette],
                    color: color.inverted[palette] ? theme.light : theme.dark,
                    padding: theme.space1,
                    margin: theme.space1,
                    minWidth: '10%'
                  })}
                  color={color}
                  palette={i}
                  key={color.palette[i]}
                  render={({ className }) => (
                    <div className={className}>
                      {i}: {palette}
                    </div>
                  )}
                />
              ))}
            </div>
          )}
        />
      ))}
    </FelaComponent>
  </Provider>
</Playground>
